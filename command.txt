section .data
    date_msg db 'Current date: ', 0
    time_msg db 'Current time: ', 0
    date_str db 'DD-MM-YYYY', 0x0a, 0x0d, 0  ; Для хранения даты в формате DD-MM-YYYY
    time_str db 'HH:MM:SS', 0x0a, 0x0d, 0     ; Для хранения времени в формате HH:MM:SS

section .text
    global _start

_start:
    ; Получение текущей даты
    mov ah, 0x2A          ; Функция получения даты
    int 0x21              ; Вызов прерывания DOS
    ; Дата в регистре:
    ; CX - год (с 1900)
    ; DH - месяц
    ; DL - день

    ; Форматирование даты
    mov bx, date_str
    mov [bx], dl         ; День
    call ConvertToString  ; Преобразование дня в строку
    mov byte [bx + 2], '-' ; Разделитель
    mov [bx + 3], dh     ; Месяц
    call ConvertToString  ; Преобразование месяца в строку
    mov byte [bx + 5], '-' ; Разделитель
    add cx, 1900         ; Преобразование года
    mov [bx + 6], cl     ; Год (младший байт)
    mov [bx + 7], ch     ; Год (старший байт)

    ; Вывод сообщения о дате
    ; mov ah, 0x09
    ; lea dx, date_msg
    ; int 0x21
    ; mov ah, 0x09
    ; lea dx, date_str
    ; int 0x21

    ; Получение текущего времени
    mov ah, 0x2C          ; Функция получения времени
    int 0x21              ; Вызов прерывания DOS
    ; Время в регистре:
    ; CH - часы
    ; CL - минуты
    ; DH - секунды

    ; Форматирование времени
    mov bx, time_str
    mov [bx], ch         ; Часы
    call ConvertToString  ; Преобразование часов в строку
    mov byte [bx + 2], ':' ; Разделитель
    mov [bx + 3], cl     ; Минуты
    call ConvertToString  ; Преобразование минут в строку
    mov byte [bx + 5], ':' ; Разделитель
    mov [bx + 6], dh     ; Секунды
    call ConvertToString  ; Преобразование секунд в строку

    ; Вывод сообщения о времени
    ; mov ah, 0x09
    ; lea dx, time_msg
    ; int 0x21
    ; mov ah, 0x09
    ; lea dx, time_str
    ; int 0x21

    ; Завершение программы
    mov ax, 0x4C00
    int 0x21

; Подпрограмма для преобразования числа в строку
ConvertToString:
    ; Вход: AL - число (0-99)
    ; Выход: строка в памяти
    push ax
    xor cx, cx           ; Обнуляем CX для подсчета цифр
    mov bx, 10           ; Делитель

.next_digit:
    xor dx, dx           ; Обнуляем DX перед делением
    div bx                ; AX / 10, результат в AL, остаток в AH
    push dx              ; Сохраняем остаток (цифру)
    inc cx               ; Увеличиваем счетчик цифр
    test ax, ax          ; Проверяем, не ноль ли результат
    jnz .next_digit      ; Если не ноль, продолжаем

.print_digits:
    pop dx               ; Извлекаем цифру
    add dl, '0'          ; Преобразуем в ASCII
    mov [bx], dl         ; Сохраняем в строку
    inc bx               ; Переход к следующему символу
    loop .print_digits    ; Печатаем все цифры

    mov byte [bx], 0     ; Завершаем строку нулем
    pop ax
    ret
